variable_declaration_list:
          declaration                                 { printf("variable_declaration_list\n"); }
        | variable_declaration_list declaration       { printf("variable_declaration_list\n"); }

declaration:
          variable_declaration
        | object_creation

variable_declaration: 
                     DATATYPE IDENT SEMICOLON               { printf("variable_declaration\n"); }
                    | modifier DATATYPE IDENT SEMICOLON     { printf("variable_declaration\n"); }
        ;

object_creation:
                modifier CLASS_NAME IDENT EQUALS NEW CLASS_NAME LPAR RPAR SEMICOLON     { printf("object_creation\n"); }
                | CLASS_NAME IDENT EQUALS NEW CLASS_NAME LPAR RPAR SEMICOLON            { printf("object_creation\n"); }
